> 示例接口：商品订单创建
> 
> 接口地址：`POST /api/v1/orders`
> 
> 功能：创建商品订单，包含商品信息、收货地址、支付方式

## 架构数据流向

```
L5 (Test Case) 
    ↓ 调用
L4 (Scenario) - 业务编排
    ↓ 使用 L3 生成/操作数据
L3 (Entity) - 数据模型（Pydantic）+ 业务逻辑
    ↓ 传给
L2 (Builder) - 接收 Entity，调用 L1
    ↓ 传给
L1 (API) - HTTP 通信
    ↓ HTTP
   后端服务
```

# -*- coding: utf-8 -*-
"""订单管理 API - L1 接口层"""
from typing import Dict, Any, Optional
from core.api import APIClient


class OrderAPI:
    """订单 API 封装
    
    职责：只关心 HTTP 通信，接收 Dict，返回 Dict
    """
    
    def __init__(self, client: APIClient):
        self.client = client
        self.base_path = "/api/v1/orders"
    
    def create_order(self, payload: Dict[str, Any]) -> Dict[str, Any]:
        """
        创建订单
        
        @param payload: 订单数据字典（由 L3 Entity.to_api_payload() 提供）
        @return: API 响应字典
        """
        response = self.client.post(self.base_path, json=payload)
        return response.json()
    
    def get_order(self, order_id: str) -> Dict[str, Any]:
        """查询订单详情"""
        response = self.client.get(f"{self.base_path}/{order_id}")
        return response.json()
    
    def cancel_order(self, order_id: str) -> Dict[str, Any]:
        """取消订单"""
        response = self.client.post(f"{self.base_path}/{order_id}/cancel")
        return response.json()
    
    def pay_order(self, order_id: str, payment_method: str) -> Dict[str, Any]:
        """支付订单"""
        payload = {"payment_method": payment_method}
        response = self.client.post(f"{self.base_path}/{order_id}/pay", json=payload)
        return response.json()
```

# -*- coding: utf-8 -*-
"""订单构造器 - L2 构造器层"""
from typing import Dict, Any, Optional
from core.base import BaseBuilder
from auto_tests.demo_project.api_manager import demo_project
from auto_tests.demo_project.data_factory.entities.order_entity import OrderEntity


class OrderBuilder(BaseBuilder):
    """
    订单构造器
    
    职责：
    1. 接收 L3 Entity
    2. 调用 L1 API 创建订单
    3. 更新 Entity 状态
    4. 管理订单生命周期
    """
    
    def __init__(self, token: str = None, factory=None):
        super().__init__(token, factory)
    
    def create_order(self, entity: OrderEntity) -> Optional[OrderEntity]:
        """
        创建订单
        
        @param entity: L3 OrderEntity（包含订单数据）
        @return: 更新后的 OrderEntity（包含 order_id, status 等）
        """
        # 1. 使用 Entity 的序列化方法获取 API 参数
        payload = entity.to_api_payload()
        
        # 2. 调用 L1 API
        result = demo_project.order.create_order(payload)
        
        # 3. 更新 Entity 状态
        if result.get("code") == 200:
            data = result.get("data", {})
            entity.order_id = data.get("order_id", "")
            entity.status = data.get("status", "pending_payment")
            entity.total_amount = data.get("total_amount", 0.0)
            entity.payment_expire_at = data.get("payment_expire_at", "")
            entity.created_at = data.get("created_at", "")
            entity.mark_as_saved(entity.order_id)
            
            # 4. 记录创建的 ID 用于清理
            self._created_ids.append(entity.order_id)
            return entity
        
        return None
    
    def create_paid_order(self, entity: OrderEntity) -> Optional[OrderEntity]:
        """快捷方法：创建已支付订单"""
        order = self.create_order(entity)
        if order and order.order_id:
            # 调用支付
            result = demo_project.order.pay_order(order.order_id, order.payment_method)
            if result.get("code") == 200:
                order.status = "paid"
        return order
    
    def create_shipped_order(self, entity: OrderEntity) -> Optional[OrderEntity]:
        """快捷方法：创建已发货订单"""
        order = self.create_paid_order(entity)
        if order and order.order_id:
            # 模拟发货
            demo_project.order.ship_order(order.order_id)
            order.status = "shipped"
        return order
    
    def cleanup(self):
        """清理创建的订单"""
        for order_id in self._created_ids:
            try:
                demo_project.order.cancel_order(order_id)
            except Exception:
                pass
        self._created_ids.clear()

# -*- coding: utf-8 -*-
"""订单实体 - L3 实体层"""
from typing import Dict, Any, Optional
from datetime import datetime
from pydantic import BaseModel, Field, field_validator


class AddressModel(BaseModel):
    """地址子模型"""
    name: str = Field(default="张三", description="收货人姓名")
    phone: str = Field(default="13800138000", description="收货人电话")
    province: str = Field(default="广东省", description="省份")
    city: str = Field(default="深圳市", description="城市")
    district: str = Field(default="南山区", description="区县")
    detail: str = Field(default="科技园南路88号", description="详细地址")
    
    def validate(self) -> bool:
        """验证地址信息完整性"""
        return all([self.name, self.phone, self.province, self.city, self.detail])


class OrderEntity(BaseModel):
    """
    订单实体 - 使用 Pydantic
    
    职责：
    1. 数据模型定义（字段、类型、默认值）
    2. 数据验证（Pydantic 自动验证）
    3. 业务逻辑（状态检查、计算等）
    4. 序列化/反序列化
    """
    
    # 订单基本信息
    order_id: str = Field(default="", description="订单ID")
    product_id: int = Field(default=1001, gt=0, description="商品ID")
    quantity: int = Field(default=1, gt=0, description="购买数量")
    total_amount: float = Field(default=0.0, ge=0, description="订单总金额")
    status: str = Field(default="pending_payment", description="订单状态")
    
    # 收货地址
    address: AddressModel = Field(default_factory=AddressModel, description="收货地址")
    
    # 支付信息
    payment_method: str = Field(default="alipay", description="支付方式")
    payment_expire_at: str = Field(default="", description="支付过期时间")
    
    # 其他
    remark: str = Field(default="", description="备注")
    created_at: str = Field(default="", description="创建时间")
    
    # 内部状态标记
    _is_saved: bool = False
    _entity_id: str = ""
    
    @field_validator('product_id', 'quantity')
    @classmethod
    def validate_positive(cls, v: int) -> int:
        """验证正整数"""
        if v <= 0:
            raise ValueError('必须是正整数')
        return v
    
    # ==================== 业务行为 ====================
    
    def calculate_total(self, unit_price: float) -> float:
        """计算订单总价"""
        self.total_amount = unit_price * self.quantity
        return self.total_amount
    
    def can_pay(self) -> bool:
        """检查是否可以支付"""
        return self.status == "pending_payment"
    
    def can_cancel(self) -> bool:
        """检查是否可以取消"""
        return self.status in ["pending_payment", "paid"]
    
    def is_expired(self) -> bool:
        """检查支付是否过期"""
        if not self.payment_expire_at:
            return False
        try:
            expire_time = datetime.fromisoformat(self.payment_expire_at.replace("Z", "+00:00"))
            return datetime.now(expire_time.tzinfo) > expire_time
        except:
            return False
    
    def validate_order(self) -> bool:
        """验证订单数据完整性"""
        if self.product_id <= 0 or self.quantity <= 0:
            return False
        if not self.address.validate():
            return False
        return True
    
    # ==================== 序列化方法 ====================
    
    def to_api_payload(self) -> Dict[str, Any]:
        """
        转换为 API 请求参数
        
        这是唯一的数据定义源头，L1/L2 都使用这个方法
        """
        return {
            "product_id": self.product_id,
            "quantity": self.quantity,
            "address": self.address.model_dump(),
            "payment_method": self.payment_method,
            "remark": self.remark
        }
    
    @classmethod
    def from_api_response(cls, data: Dict[str, Any]) -> "OrderEntity":
        """从 API 响应创建实体"""
        return cls(
            order_id=data.get("order_id", ""),
            status=data.get("status", "pending_payment"),
            total_amount=data.get("total_amount", 0.0),
            payment_expire_at=data.get("payment_expire_at", ""),
            created_at=data.get("created_at", "")
        )
    
    # ==================== 工厂方法 ====================
    
    @classmethod
    def with_product(cls, product_id: int, quantity: int = 1) -> "OrderEntity":
        """创建指定商品的订单"""
        return cls(product_id=product_id, quantity=quantity)
    
    @classmethod
    def with_address(cls, address: Dict[str, str]) -> "OrderEntity":
        """创建指定地址的订单"""
        return cls(address=AddressModel(**address))
    
    @classmethod
    def default(cls) -> "OrderEntity":
        """创建默认订单"""
        return cls()
    
    # ==================== 生命周期管理 ====================
    
    def mark_as_saved(self, entity_id: str):
        """标记为已保存"""
        self._is_saved = True
        self._entity_id = entity_id
    
    def is_saved(self) -> bool:
        """检查是否已保存"""
        return self._is_saved
    
    def model_dump(self, **kwargs) -> Dict[str, Any]:
        """重写 model_dump，确保地址也被序列化"""
        data = super().model_dump(**kwargs)
        data['address'] = self.address.model_dump(**kwargs)
        return data
```

## L4: 场景层 (Scenario)

**文件**: `data_factory/scenarios/create_order_scenario.py`

```python
# -*- coding: utf-8 -*-
"""创建订单场景 - L4 场景层"""
from typing import Dict, Any, Optional, List
from core.base import BaseScenario, ScenarioResult
from auto_tests.demo_project.data_factory.entities.order_entity import OrderEntity, AddressModel
from auto_tests.demo_project.data_factory.builders.order_builder import OrderBuilder


class CreateOrderScenario(BaseScenario):
    """
    创建订单场景
    
    封装完整的订单创建流程：
    1. 使用 L3 Entity 创建数据模型
    2. 使用 L3 的业务逻辑进行验证
    3. 调用 L2 Builder 创建订单
    4. 更新 L3 Entity 状态
    
    数据流向：L4 → L3 → L2 → L1
    """
    
    # 场景依赖声明
    dependencies = []  # 创建订单无依赖
    creates = [OrderEntity]  # 场景创建 OrderEntity
    
    def __init__(self, token: str = None, factory=None):
        super().__init__(token, factory)
        self.order_builder = OrderBuilder(token, factory)
    
    def execute(
        self,
        product_id: int = 1001,
        quantity: int = 1,
        address: Dict[str, str] = None,
        payment_method: str = "alipay",
        remark: str = "",
        skip_stock_check: bool = False
    ) -> ScenarioResult:
        """
        执行创建订单场景
        
        @param product_id: 商品ID
        @param quantity: 数量
        @param address: 收货地址
        @param payment_method: 支付方式
        @param remark: 备注
        @param skip_stock_check: 是否跳过库存检查
        @return: 场景执行结果
        """
        result = ScenarioResult()
        
        # 1. 使用 L3 创建数据模型
        address_model = AddressModel(**(address or {}))
        order_entity = OrderEntity(
            product_id=product_id,
            quantity=quantity,
            address=address_model,
            payment_method=payment_method,
            remark=remark
        )
        
        # 2. 使用 L3 的业务逻辑进行验证
        if not order_entity.validate_order():
            result.add_error("订单数据验证失败")
            return result
        
        # 3. 检查库存（模拟）
        if not skip_stock_check:
            stock = self._check_stock(product_id)
            if stock < quantity:
                result.add_error(f"库存不足，当前库存: {stock}")
                return result
        
        # 4. 调用 L2 Builder 创建订单（传递 L3 Entity）
        created_order = self.order_builder.create_order(order_entity)
        
        if not created_order:
            result.add_error("订单创建失败")
            return result
        
        # 5. 注册实体到场景
        self.register_entity(created_order)
        result.add_entity("order", created_order)
        result.data = created_order.model_dump()
        result.success = True
        
        return result
    
    def _check_stock(self, product_id: int) -> int:
        """检查商品库存（模拟）"""
        # 实际项目中调用库存接口
        return 100
    
    # ==================== 变体矩阵支持 ====================
    
    @classmethod
    def variants(cls) -> List[Dict[str, Any]]:
        """生成场景变体，用于参数化测试"""
        variants = []
        products = [1001, 1002, 1003]
        quantities = [1, 2, 5]
        payments = ["alipay", "wechat", "credit_card"]
        
        for product in products:
            for quantity in quantities:
                for payment in payments:
                    variants.append({
                        "product_id": product,
                        "quantity": quantity,
                        "payment_method": payment,
                        "description": f"商品{product}_数量{quantity}_{payment}"
                    })
        
        return variants


class OrderPaymentScenario(BaseScenario):
    """
    订单支付场景
    
    依赖：已创建的订单（OrderEntity）
    """
    
    dependencies = [OrderEntity]
    
    def __init__(self, token: str = None, factory=None):
        super().__init__(token, factory)
        self.order_builder = OrderBuilder(token, factory)
    
    def execute(
        self,
        order_id: str = None,
        payment_method: str = None
    ) -> ScenarioResult:
        """执行支付场景"""
        result = ScenarioResult()
        
        # 1. 获取 L3 OrderEntity
        order = self.get_entity(OrderEntity, order_id)
        if not order:
            result.add_error("订单不存在")
            return result
        
        # 2. 使用 L3 的业务逻辑检查
        if not order.can_pay():
            result.add_error(f"订单状态不允许支付: {order.status}")
            return result
        
        if order.is_expired():
            result.add_error("订单支付已过期")
            return result
        
        # 3. 执行支付（调用 L2）
        pay_result = self.order_builder.pay_order(order, payment_method or order.payment_method)
        
        if pay_result:
            result.success = True
            result.data = {"order_id": order.order_id, "status": order.status}
        else:
            result.add_error("支付失败")
        
        return result
```

import pytest

from core.base import UnitTest
from auto_tests.demo_project.data_factory.entities.order_entity import OrderEntity

@allure.story("订单完整生命周期")

    @allure.title("完整订单生命周期：创建-支付-发货-完成")
    def test_order_full_lifecycle(self, api_client):

        # 1. 创建订单
        order = OrderEntity.with_product(product_id=1001, quantity=2)
        builder = OrderBuilder()
        created = builder.create_order(order)
        
        assert created.status == "pending_payment"
        
        # 2. 支付订单
        api_client.order.pay_order(created.order_id, "alipay")
        
        # 3. 发货
        api_client.order.ship_order(created.order_id)
        
        # 4. 完成
        api_client.order.complete_order(created.order_id)
        
        # 5. 验证最终状态
        final = api_client.order.get_order(created.order_id)
        assert final["data"]["status"] == "completed"

# -*- coding: utf-8 -*-
"""订单场景化测试 - L5 用例层"""
import pytest
import allure
from core.base import UnitTest
from auto_tests.demo_project.data_factory.scenarios import CreateOrderScenario


@allure.feature("订单管理")
@allure.story("订单场景")
class TestOrderScenario(UnitTest):
    """订单场景化测试 - 使用 L4 Scenario"""
    
    @allure.title("使用 Scenario 创建订单")
    def test_create_order_with_scenario(self, test_context):
        """通过 L4 Scenario 创建订单"""
        # L5 调用 L4，L4 使用 L3 并传给 L2 → L1
        result = test_context.scenario(
            CreateOrderScenario,
            product_id=1001,
            quantity=2,
            payment_method="alipay"
        )
        
        assert result.success
        
        # 获取 L3 Entity
        order = result.get_entity("order")
        assert order.order_id
        assert order.status == "pending_payment"
        
        # 使用 L3 业务逻辑验证
        assert order.can_pay()
        assert not order.is_expired()
    
    @allure.title("场景变体矩阵测试")
    @pytest.mark.parametrize("variant", CreateOrderScenario.variants())
    def test_create_order_variants(self, test_context, variant):
        """测试多种订单变体组合"""
        result = test_context.scenario(CreateOrderScenario, **variant)
        
        assert result.success, f"变体失败: {variant.get('description')}"
        
        order = result.get_entity("order")
        assert order.total_amount > 0
        assert order.validate_order()

## 总结

### 五层调用关系

```
L5 (Test Case)
    ↓ 调用
L4 (Scenario) - 业务编排
    ↓ 使用 L3 生成/操作数据
L3 (OrderEntity) - Pydantic 数据模型 + 业务逻辑
    ↓ to_api_payload()
L2 (OrderBuilder) - 接收 Entity，调用 L1
    ↓ 传递 Dict
L1 (OrderAPI) - HTTP 通信
    ↓ HTTP POST
   后端服务
```

### 各层职责

| 层级 | 文件 | 核心职责 | 输入 | 输出 |
|------|------|----------|------|------|
| L1 | `api_manager/order.py` | HTTP 请求封装 | `Dict` | `Dict` |
| L2 | `builders/order_builder.py` | 接收 Entity，调用 L1 | `OrderEntity` | `OrderEntity` |
| L3 | `entities/order_entity.py` | Pydantic 模型 + 业务逻辑 | 参数 | `OrderEntity` |
| L4 | `scenarios/create_order_scenario.py` | 业务编排，使用 L3 | 参数 | `ScenarioResult` |
| L5 | `test_cases/test_order.py` | 测试执行与断言 | - | 测试结果 |

### 关键设计点

1. **L3 Entity 是唯一数据定义源头**：字段只在 L3 定义一次，通过 `to_api_payload()` 提供给 L1/L2
2. **Pydantic 提供数据验证**：自动验证字段类型和约束
3. **业务逻辑集中在 L3**：状态检查、计算等方法在 Entity 中实现
4. **L4 负责业务编排**：组合多个 Entity 完成复杂业务流程
5. **数据流向清晰**：L5→L4→L3→L2→L1，每层职责单一
